Key SQL Queries for SCSIT DAVV
-- 1. Attendance % per student per subject (MCA 2nd Sem Batch A) SELECT s.roll_no, s.name, sub.subject_code, sub.subject_name, COUNT(*) AS total, SUM(a.status='P') AS present, ROUND(SUM(a.status='P') * 100.0 / COUNT(*), 2) AS attendance_pct FROM attendance a JOIN students s ON a.student_id = s.student_id JOIN teacher_assignments ta ON a.assign_id = ta.assign_id JOIN subjects sub ON ta.subject_id = sub.subject_id JOIN sections sec ON ta.section_id = sec.section_id WHERE sec.batch_name = 'A' AND sec.semester = 2 GROUP BY s.student_id, sub.subject_id ORDER BY attendance_pct ASC; -- 2. Students below 75% (Detention candidates) SELECT s.roll_no, s.name, sub.subject_name, ROUND(SUM(a.status='P') * 100.0 / COUNT(*), 2) AS pct FROM attendance a JOIN students s ON a.student_id = s.student_id JOIN teacher_assignments ta ON a.assign_id = ta.assign_id JOIN subjects sub ON ta.subject_id = sub.subject_id GROUP BY s.student_id, sub.subject_id HAVING pct < 75 ORDER BY pct; -- 3. Teacher-wise classes conducted SELECT t.name, t.teacher_type, COUNT(DISTINCT ta.subject_id) AS subjects, COUNT(DISTINCT ta.section_id) AS sections, COUNT(a.attend_id) AS classes_conducted FROM teachers t LEFT JOIN teacher_assignments ta ON t.teacher_id = ta.teacher_id LEFT JOIN attendance a ON ta.assign_id = a.assign_id AND a.marked_by = t.teacher_id GROUP BY t.teacher_id ORDER BY classes_conducted DESC; -- 4. Today's attendance for a section SELECT s.roll_no, s.name, sub.subject_code, a.status FROM attendance a JOIN students s ON a.student_id = s.student_id JOIN teacher_assignments ta ON a.assign_id = ta.assign_id JOIN subjects sub ON ta.subject_id = sub.subject_id WHERE a.date = CURDATE() AND ta.section_id = 7 ORDER BY s.roll_no;
